  
#include <stdio.h>

int num(int n);

int main(void)
{
    int i;
    printf("Fibonacci sequence: ");
    for(i = 1; i <= 6; i++)
        printf("%d ", num(i-1));
    printf("\n");

    return 0;
}

int num(int n)
{
    if(n == 0)
        return 0;
    else if(n == 1)
        return 1;
    return num(n-2) + num(n-1);
}
